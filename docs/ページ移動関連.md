## 新しいページが下からゆっくり登場バージョン

import 'package:flutter/material.dart';

void main() => runApp(const MaterialApp(
  home: PageA(),
  debugShowCheckedModeBanner: false,
));

// --- ページA (出発) ---
class PageA extends StatelessWidget {
  const PageA({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: const Color(0xFF121212),
      body: Center(
        child: ElevatedButton(
          onPressed: () => Navigator.of(context).push(speakRoute(const PageB())),
          child: const Text("スロー演出を開始"),
        ),
      ),
    );
  }
}

// --- ページB (ここをいじる！) ---
class PageB extends StatelessWidget {
  const PageB({super.key});

  @override
  Widget build(BuildContext context) {
    // ページ遷移の「animation」を子要素に伝搬させる
    final animation = ModalRoute.of(context)!.animation!;

    return Scaffold(
      backgroundColor: Colors.black,
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            // 1. テキスト：早めに、標準的な速さで
            SpeakReveal(
              animation: animation,
              start: 0.5, // 30%から開始
              child: const Text(
                "まず、メッセージが届く",
                style: TextStyle(color: Colors.white, fontSize: 24),
              ),
            ),
            const SizedBox(height: 50),
            
            // 2. ボタン：もっと遅く、さらに「ゆっくり」登場させる
            SpeakReveal(
              animation: animation,
              start: 0.6, // 60%から開始（かなり遅れて出る）
              // 終了を 1.0 にすることで、残りの40%の時間を使って「じわ〜」っと出す
              child: ElevatedButton(
                style: ElevatedButton.styleFrom(
                  padding: const EdgeInsets.symmetric(horizontal: 40, vertical: 15),
                ),
                onPressed: () => Navigator.pop(context),
                child: const Text("もっとゆっくり現れるボタン"),
              ),
            ),
          ],
        ),
      ),
    );
  }
}

// --- 個別にスピード調整できる魔法のコンポーネント ---
class SpeakReveal extends StatelessWidget {
  final Animation<double> animation;
  final Widget child;
  final double start; // 開始タイミング (0.0 〜 1.0)
  final double end;   // 終了タイミング (0.0 〜 1.0)

  const SpeakReveal({
    required this.animation, 
    required this.child, 
    this.start = 0.2, 
    this.end = 1.0, // ここを大きく取るとゆっくりになる
    super.key
  });

  @override
  Widget build(BuildContext context) {
    // SpeakRevealウィジェットの中の修正
final curvedAnimation = CurvedAnimation(
      parent: animation,
      curve: Interval(
        0.2,             // ここが開始（start）
        1.0,             // ここが終了（end）
        curve: Curves.linear, // ここで質感を指定
      ),
    );

    return AnimatedBuilder(
      animation: curvedAnimation,
      builder: (context, _) {
        return Opacity(
          opacity: curvedAnimation.value,
          child: Transform.translate(
            // 距離を 40px に増やすと、より遠くからゆっくり来る感じが出る
            offset: Offset(0, 80 * (1 - curvedAnimation.value)),
            child: child,
          ),
        );
      },
    );
  }
}

// --- 共通のルート定義 ---
Route speakRoute(Widget page) {
  return PageRouteBuilder(
    transitionDuration: const Duration(milliseconds: 2000), // 全体は2秒
    pageBuilder: (context, animation, secondaryAnimation) => page,
    transitionsBuilder: (context, animation, secondaryAnimation, child) {
      return FadeTransition(opacity: animation, child: child);
    },
  );
}



## ポップアップから順番に時間差で表示されるバージョン

import 'dart:ui'; // ぼかし効果(ImageFilter)に必要

void main() => runApp(const MaterialApp(

// --- ページA (出発：背景として残るページ) ---
class PageA extends StatelessWidget {

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Container(
        width: double.infinity,
        decoration: const BoxDecoration(
          image: DecorationImage(

            fit: BoxFit.cover,
          ),
        ),
        child: Center(
          child: ElevatedButton(

            child: const Text("スタッガード遷移を開始"),

  }
}

// --- ページB (到着：時間差で要素が出る) ---
class PageB extends StatelessWidget {
  const PageB({super.key});

  @override
  Widget build(BuildContext context) {
    final animation = ModalRoute.of(context)!.animation!;

    return Scaffold(
      backgroundColor: Colors.transparent, // 背景を透明にして下のページを見せる
      body: Stack(
        children: [
          // 1. 背景のぼかし (前のページを透かしつつボカす)
          AnimatedBuilder(
            animation: animation,
            builder: (context, child) {
              return BackdropFilter(
                filter: ImageFilter.blur(
                  sigmaX: 10 * animation.value, 
                  sigmaY: 10 * animation.value
                ),
                child: Container(color: Colors.black.withOpacity(0.4 * animation.value)),
              );
            },
          ),
          
          // 2. コンテンツ (時間差で順番に登場)
          Center(
            child: Padding(
              padding: const EdgeInsets.symmetric(horizontal: 30),
              child: Column(
                mainAxisSize: MainAxisSize.min,
                children: [
                  // --- ここからスタッガード（時間差）演出 ---
                  
                  // A. ポップアップの土台 (一番先に出る)
                  SpeakReveal(
                    animation: animation,
                    start: 0.1, // 10%から
                    child: Container(
                      padding: const EdgeInsets.all(24),
                      decoration: BoxDecoration(
                        color: Colors.white,
                        borderRadius: BorderRadius.circular(24),
                      ),
                      child: Column(
                        children: [
                          // B. ポップアップ内のタイトル (少し遅れて)
                          SpeakReveal(
                            animation: animation,
                            start: 0.3, // 30%から
                            child: const Text("タイトル", 
                              style: TextStyle(fontSize: 24, fontWeight: FontWeight.bold, color: Colors.black)),
                          ),
                          const SizedBox(height: 16),
                          
                          // C. 説明テキスト (さらに遅れて)
                          SpeakReveal(
                            animation: animation,
                            start: 0.5, // 50%から
                            child: const Text("ここに説明のテキストが順番に流れるように現れます。", 
                              textAlign: TextAlign.center,
                              style: TextStyle(fontSize: 16, color: Colors.black87)),
                          ),
                          const SizedBox(height: 24),
                          
                          // D. ボタン (最後に出る)
                          SpeakReveal(
                            animation: animation,
                            start: 0.7, // 70%から
                            child: SizedBox(
                              width: double.infinity,
                              child: ElevatedButton(
                                onPressed: () => Navigator.pop(context),
                                child: const Text("了解しました"),
                              ),
                            ),
                          ),
                        ],
                      ),
                    ),
                  ),
                ],
              ),
            ),
          ),
        ],
      ),
    );
  }
}

// --- 以下、使い回し用テンプレートパーツ ---

class SpeakReveal extends StatelessWidget {
  final Animation<double> animation;
  final Widget child;
  final double start;
  final double end;

  const SpeakReveal({required this.animation, required this.child, this.start = 0.0, this.end = 1.0, super.key});

  @override
  Widget build(BuildContext context) {
    final curvedAnimation = CurvedAnimation(
      parent: animation,
      curve: Interval(start, end, curve: Curves.easeOutCubic),
    );

    return AnimatedBuilder(
      animation: curvedAnimation,
      builder: (context, _) {
        return Opacity(
          opacity: curvedAnimation.value,
          child: Transform.translate(
            offset: Offset(0, 20 * (1 - curvedAnimation.value)),
            child: child,
          ),
        );
      },
    );
  }
}

Route speakBlurRoute(Widget page) {
  return PageRouteBuilder(
    opaque: false, // ここをfalseにすることで、前のページが消えずに残る！
    transitionDuration: const Duration(milliseconds: 2000),
    pageBuilder: (context, animation, secondaryAnimation) => page,
    transitionsBuilder: (context, animation, secondaryAnimation, child) {
      return FadeTransition(opacity: animation, child: child);
    },
  );
}

このテンプレートの「調整ポイント」
残像の濃さ:
Container(color: Colors.black.withOpacity(0.4 * animation.value))
この 0.4 を 0.7 にするとより暗く、0.2 にするとより前のページがはっきり見えます。

ぼかしの強さ:
sigmaX: 10 * animation.value
この 10 を 20 にすると、前のページがもっとトロトロにボケます。

時間差の黄金比:
今回のコードでは 0.1 → 0.3 → 0.5 → 0.7 と 0.2刻み で設定しています。

もっと「矢継ぎ早」に出したいなら：0.1 → 0.2 → 0.3 → 0.4

もっと「一つ一つをしっかり見せたい」なら：0.1 → 0.4 → 0.7 → 0.9


## ポップアップから順番に表示され最後に自然と消えるバージョン

import 'package:flutter/material.dart';
import 'dart:ui';

void main() => runApp(const MaterialApp(home: PageA(), debugShowCheckedModeBanner: false));

// --- ★ここ★ 演出の「黄金比」設定表 ---
class BridgeConfig {
  static const showDuration = 5000;    // ポップアップが表示されている時間（ms）
  static const fadeOutSpeed = 800;     // 消える時のスピード（ms）
  static const blurExitSpeed = 1200;   // 背景のボケが晴れるスピード（ms）
}

class PageA extends StatelessWidget {
  const PageA({super.key});
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Center(
        child: ElevatedButton(
          onPressed: () => Navigator.of(context).push(speakBlurRoute(const BridgePage())),
          child: const Text("演出開始"),
        ),
      ),
    );
  }
}

class BridgePage extends StatefulWidget {
  const BridgePage({super.key});
  @override
  State<BridgePage> createState() => _BridgePageState();
}

class _BridgePageState extends State<BridgePage> {
  bool _isExiting = false;

  @override
  void initState() {
    super.initState();
    
    // ★ここ★ 表示時間の制御
    Future.delayed(const Duration(milliseconds: BridgeConfig.showDuration), () {
      if (mounted) setState(() => _isExiting = true);
    });

    // 消え終わるのを待ってから画面を閉じる
    Future.delayed(Duration(milliseconds: BridgeConfig.showDuration + BridgeConfig.fadeOutSpeed), () {
       if (mounted) Navigator.pop(context);
    });
  }

  @override
  Widget build(BuildContext context) {
    final animation = ModalRoute.of(context)!.animation!;

    return Scaffold(
      backgroundColor: Colors.transparent,
      body: Stack(
        children: [
          // 背景ぼかしの消えるスピード設定
          AnimatedBuilder(
            animation: animation,
            builder: (context, child) {
              double blurAmount = _isExiting ? 0.0 : 15.0;
              return BackdropFilter(
                filter: ImageFilter.blur(sigmaX: blurAmount, sigmaY: blurAmount),
                child: AnimatedContainer(
                  // ★ここ★ ボケが晴れるスピード
                  duration: const Duration(milliseconds: BridgeConfig.blurExitSpeed),
                  color: Colors.black.withOpacity(_isExiting ? 0.0 : 0.5),
                ),
              );
            },
          ),

          Center(
            child: AnimatedOpacity(
              opacity: _isExiting ? 0.0 : 1.0,
              // ★ここ★ ポップアップが消えるスピード
              duration: const Duration(milliseconds: BridgeConfig.fadeOutSpeed),
              curve: Curves.easeInOut,
              child: Container(
                width: 300,
                padding: const EdgeInsets.all(30),
                decoration: BoxDecoration(
                  color: Colors.white,
                  borderRadius: BorderRadius.circular(30),
                  boxShadow: [BoxShadow(color: Colors.black26, blurRadius: 20)],
                ),
                child: Column(
                  mainAxisSize: MainAxisSize.min,
                  children: [
                    // --- ★ここ★ 登場の順番（Interval）の設定 ---
                    
                    // 1. アイコン（10%〜50%の間で登場）
                    SpeakReveal(
                      animation: animation,
                      start: 0.1, 
                      child: const Icon(Icons.auto_awesome, size: 50, color: Colors.amber),
                    ),
                    const SizedBox(height: 20),
                    
                    // 2. タイトル（30%〜70%の間で登場）
                    SpeakReveal(
                      animation: animation,
                      start: 0.5, 
                      child: const Text("素晴らしい！", style: TextStyle(fontSize: 24, fontWeight: FontWeight.bold)),
                    ),
                    const SizedBox(height: 10),
                    
                    // 3. 説明文（50%〜90%の間で登場）
                    SpeakReveal(
                      animation: animation,
                      start: 0.9, 
                      child: const Text("この調子で次のステップへ進みましょう。", textAlign: TextAlign.center),
                    ),
                  ],
                ),
              ),
            ),
          ),
        ],
      ),
    );
  }
}

// --- 登場のスピードと時間差を管理する部品 ---
class SpeakReveal extends StatelessWidget {
  final Animation<double> animation;
  final Widget child;
  final double start; // ★開始タイミング★

  const SpeakReveal({required this.animation, required this.child, required this.start, super.key});

  @override
  Widget build(BuildContext context) {
    // startから「+0.4」秒間（割合）かけて登場させる設定
    final curvedAnimation = CurvedAnimation(
      parent: animation,
      curve: Interval(start, (start + 0.6).clamp(0.0, 1.0), curve: Curves.easeOutCubic),
    );

    return AnimatedBuilder(
      animation: curvedAnimation,
      builder: (context, _) {
        return Opacity(
          opacity: curvedAnimation.value,
          child: Transform.translate(
            offset: Offset(0, 20 * (1 - curvedAnimation.value)),
            child: child,
          ),
        );
      },
    );
  }
}
